cmake_minimum_required(VERSION 3.15)
project(svgtest LANGUAGES CXX)

if(WIN32)
  file(GLOB SRC_FILES *.cpp *.h *.rc ${CMAKE_SOURCE_DIR}/*.xpm)
else()
  file(GLOB SRC_FILES *.cpp *.h ${CMAKE_SOURCE_DIR}/*.xpm)
endif()
if(WIN32)
  add_executable(svgtest WIN32 ${SRC_FILES})
else()
  add_executable(svgtest ${SRC_FILES})
endif()
target_link_libraries(svgtest PRIVATE ${wxWidgets_LIBRARIES})
