cmake_minimum_required(VERSION 3.15)
project(toolbar LANGUAGES CXX)

if(WIN32)
  file(GLOB SRC_FILES *.cpp *.h *.rc ${CMAKE_SOURCE_DIR}/*.xpm)
else()
  file(GLOB SRC_FILES *.cpp *.h ${CMAKE_SOURCE_DIR}/*.xpm)
endif()
if(WIN32)
  add_executable(toolbar WIN32 ${SRC_FILES})
else()
  add_executable(toolbar ${SRC_FILES})
endif()
target_link_libraries(toolbar PRIVATE ${wxWidgets_LIBRARIES})
